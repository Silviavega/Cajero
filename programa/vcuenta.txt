    private void btnAhorrosActionPerformed(java.awt.event.ActionEvent evt) {                                           
       int n= JOptionPane.showConfirmDialog(null,"Desea contituar con la transacción:");
        if (n==0) {
            Cuenta objCuenta= new Cuenta();
            objCuenta.verificarSaldo(""+Conectar.numCuenta);
            if(Cuenta.tipoCuenta==2 && Cuenta.saldo>=vRetiro.valor){    
                
                double saldoActual=Cuenta.saldo-vRetiro.valor;
                objCuenta.actualizarSaldo(Conectar.numCuenta,saldoActual );                  
               
                String descripcion="Retiro";
                if(vTipo_Servicio.tipoServicio==1){
               descripcion="Pago Servicio Telefónia "+vServicio.nombreOperadora;
                  }
                
                 if (vTipo_Servicio.tipoServicio==2){
               descripcion="Pago Servicio Básico "+vServicio2.nombreServicio;
                }
                 
                if (Menu.TipoTransferencia==3) {
                  descripcion="Transferencia al N° cuenta "+vValoryCuentaTransferir.nombreCuenta ;  
                }
                
                Transaccion objTran=new Transaccion(1, VUsuario.fechaActual(), VUsuario.horaActual(),Cuenta.saldo,vRetiro.valor, saldoActual, descripcion);
                objTran.insertarTransaccion();
                codTransaccion=objTran.maxTransaccion();
 
             //   JOptionPane.showMessageDialog(null, "Favor tome su dinero---> Transacción exitosa ");
          vTransaccion_Exitosa vTran= new vTransaccion_Exitosa();
          vTran.show();
          this.dispose();
                        
            }  else if(Cuenta.tipoCuenta==1){
             JOptionPane.showMessageDialog(null, "La tarjeta ingresada no pertenece al tipo de cuenta Ahorros ");
            }
            else if(Cuenta.saldo<=vRetiro.valor){
                JOptionPane.showMessageDialog(null, "Actualmente su saldo es insuficiente para realizar la Transacción ---> Saldo Disponible -->"+Cuenta.saldo);
           
            }
            
           // JOptionPane.showMessageDialog(null, "Tipo transferencia"+ Menu.TipoTransferencia+" saldo"+vRetiro.valor);
        
    
        }else{
            JOptionPane.showMessageDialog(null, "Usted ha cancelado su transacción");
        }
        
    }

    private void btnCorrienteActionPerformed(java.awt.event.ActionEvent evt) {

        if (n==0) {

            if(Cuenta.tipoCuenta==1 && Cuenta.saldo>=vRetiro.valor){  
                double saldoActual=Cuenta.saldo-vRetiro.valor;
                objCuenta.actualizarSaldo(Conectar.numCuenta,saldoActual );

    private void btnImprimirActionPerformed(java.awt.event.ActionEvent evt) {                                            
     // JOptionPane.showMessageDialog(null,vCuenta.codTransaccion);
     
     //    String sql="Select Usu.*, Cu.*, Tr.*, TC.* from usuario  Usu inner join  cuenta Cu on Usu.id=Cu.idUsu inner join tipo_cuenta TC on TC.id=Cu.idTipCue inner join transaccion Tr on Cu.id=Tr.idCue";
 
        String usuario="", fecha="", hora="", saldoAnterior="", retiro="", saldoDisponible="", descripcion="";
        Transaccion objtra= new Transaccion();
   
        
        
     ResultSet rs=objtra.consultarTransaccion(vCuenta.codTransaccion);
        try {
            if (rs.next()) {
                usuario=rs.getString("Usu.nombre")+" "+rs.getString("Usu.apellido");
                fecha=rs.getString("Tr.fecha");
                hora=rs.getString("Tr.hora");
                saldoAnterior=rs.getString("Tr.saldoAnterior");
                retiro=rs.getString("Tr.valorRetiro");
                saldoDisponible=rs.getString("Tr.saldoActual");
                descripcion=rs.getString("Tr.descripcion");
            
         //   JOptionPane.showMessageDialog(null, usuario+fecha+"-"+ hora+saldoAnterior+retiro+saldoDisponible+descripcion);
                    
                    }
        } catch (SQLException ex) {
            Logger.getLogger(vTransaccion_Exitosa.class.getName()).log(Level.SEVERE, null, ex);
        }
          try {
            Conectar cc= new Conectar();
            Map parametro =new HashMap();
            parametro.clear();
            parametro.put("usuario",usuario);
            parametro.put("fecha", fecha);
            parametro.put("hora", hora);
            parametro.put("saldoAnterior", saldoAnterior);
            parametro.put("retiro", retiro);
            parametro.put("saldoDisponible", saldoDisponible);
            parametro.put("descripcion", descripcion);
           // parametro.put("software", "");
            JasperReport reporCliente=JasperCompileManager.compileReport(".\\src\\comprobante\\comprobante.jrxml");
            JasperPrint print=JasperFillManager.fillReport(reporCliente,parametro,cc.abrir());
            JasperViewer visor = new JasperViewer(print, false);
            visor.setTitle("Comprobante");
            visor.show();
        } catch (JRException ex) {
        //    Logger.getLogger(vTransaccion_Exitosa.class.getName()).log(Level.SEVERE, null, ex);
        }
      
    }

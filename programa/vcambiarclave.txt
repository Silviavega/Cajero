    private void btnAceptarActionPerformed(java.awt.event.ActionEvent evt) {                                           
        if(passClave.getText().length()==4 ){
        if (passClave.getText().equals(passConfirmarClave.getText())) {
           Cuenta objCuenta= new Cuenta();
           objCuenta.actualizarClave(Conectar.numCuenta, passConfirmarClave.getText());
            
        JOptionPane.showMessageDialog(null, "SU CLAVE HA SIDO CAMBIADA EXITOSAMENTE :");
        passClave.setText("");
        passConfirmarClave.setText("");
        }else{
        JOptionPane.showMessageDialog(null, "CONTRASEÑAS NO COINCIDEN :");
        }}else
            JOptionPane.showMessageDialog(null, "FAVOR INGRESAR CUATRO DIGITOS");
        
        
    }

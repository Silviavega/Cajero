   void cargar(String valor){
       int tipoCuenta=0;
    String []titulos={"Id Usuario","Cédula", "Nombres","Apellidos","Dirección","Celular","Num Cuenta","Tipo Cuenta", " Saldo","Clave" };
    String []Registros=new String[10];
    model= new DefaultTableModel(null,titulos);
         try {
            Usuario objUser= new Usuario();
            ResultSet rs= objUser.consultarUsuario(valor);
                 while(rs.next()){
                 Registros[0]= rs.getString("Usu.id");
                  Registros[1]=rs.getString("Usu.cedula");
                  Registros[2]= rs.getString("Usu.nombre");
                  Registros[3]= rs.getString("Usu.apellido");
                  Registros[4]=rs.getString("Usu.direccion");
                  Registros[5]=rs.getString("Usu.celular");
                  Registros[6]=rs.getString("Cu.numCuenta");
                  Registros[7]=rs.getString("tipoCuenta");
                  Registros[8]=rs.getString("Cu.saldo");
                  Registros[9]=rs.getString("Cu.clave");
                  model.addRow(Registros);
                }
              tblUsuario.setModel(model);
   rs.close();
         } catch (SQLException ex) {
            Logger.getLogger(vUsuarioRegistrado.class.getName()).log(Level.SEVERE, null, ex);
        }

}

    private void btnAgregarActionPerformed(java.awt.event.ActionEvent evt) {                                           
        String id="", cedula="", nom="",ape="",direccion="",celular="", numCuenta="", tipoCuenta="", saldo="", clave="" ;
        int fila = tblUsuario.getSelectedRow();

        
        try {
            if(fila==-1)
            JOptionPane.showMessageDialog(null, "No ha seleccionado ningun Usuario :");
            else
            {
                id =  (String)tblUsuario.getValueAt(fila, 0);
                cedula=(String)tblUsuario.getValueAt(fila, 1);
                nom =  (String)tblUsuario.getValueAt(fila, 2);
                ape =  (String)tblUsuario.getValueAt(fila, 3);
                direccion=(String)tblUsuario.getValueAt(fila,4);
                celular=(String)tblUsuario.getValueAt(fila, 5);
                numCuenta =(String)tblUsuario.getValueAt(fila,6);
                tipoCuenta=(String)tblUsuario.getValueAt(fila,7);
                saldo=(String)tblUsuario.getValueAt(fila, 8);
                clave=(String)tblUsuario.getValueAt(fila, 9);
                
                VUsuario.txtCod.setText(id);
                VUsuario.txtCedula.setText(cedula);
                VUsuario.txtCedula.setEditable(false);
                VUsuario.txtNombre.setText(nom);
                VUsuario.txtApe.setText(ape);
                VUsuario.txtDireccion.setText(direccion);
                VUsuario.txtCelular.setText(celular);
                VUsuario.txtNumCuenta.setText(numCuenta);
                VUsuario.txtNumCuenta.setEditable(false);
                VUsuario.txtSaldoCuenta.setText(saldo);
                VUsuario.txtClaveCuenta.setText(clave);
                
                VUsuario.cboTipoCuenta.removeAllItems();
                VUsuario.cboTipoCuenta.addItem("Corriente");
                VUsuario.cboTipoCuenta.addItem("Ahorros");
                
                
                 this.dispose();
            }
        } catch (Exception e) {

        }
    }
